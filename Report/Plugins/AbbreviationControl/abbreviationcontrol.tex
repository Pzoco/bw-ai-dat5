%[LABEL] Abbreviation control plugin
%[VERSION] 1.0
% File: abbreviationcontrol.tex
% Automates abbreviation control through out the document

%[BEGIN] Abbreviation control macro
  % Parameters:
  %	1) command name
  %	2) long name
  %	3) acronym
  \newcommand{\acronym}[3]{%
	  \expandafter\newcommand\expandafter{\csname #1\endcsname}{#2 (#3)%
	  	\expandafter\renewcommand\expandafter{\csname #1\endcsname}{%
	  		#3%
	  	}%
	  }%
  }
%[END]

%[BEGIN] Input user-defined abbreviation
  \input{\PluginsPath/AbbreviationControl/abbreviations}
%[END]
